\subsubsection{Gestion des erreurs}
	\begin{itemize}
		\item Si l'utilisateur ne possède plus d'abonnement une erreur sera afficher lui signalant le problème. 
		\item Si la session est complète un message d'erreur sera affiché. 
	\end{itemize}
